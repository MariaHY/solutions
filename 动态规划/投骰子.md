class Solution:
    def dicesProbability(self, n: int) -> List[float]:
        dp=[[0]*(6*n+1) for _ in range(n+1)]
        for i in range(1,7):
            dp[1][i]=1
        for i in range(2,n+1):
            # print('第',i,'次投掷')
            for j in range(i,6*i+1):
                print('和为',j)
                for k in range(1,7):
                    if j-k<=0:
                        break
                    dp[i][j]+=dp[i-1][j-k]
                    # print(i-1,'次投掷出的和',j-k,'的贡献值 ',dp[i-1][j-k])
                # print('递归推导')
                # print('dp:',dp)
        res=[]            
        for i in range(n,6*n+1):
            res.append(dp[n][i]*1.0/(6**n))
        return res
