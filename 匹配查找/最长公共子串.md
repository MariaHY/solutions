### 对应字符是否相等判断公共状态
-暴力枚举
```python
import operator
class Solution:
    def LCS(self , str1 , str2 ):
        # write code here
        if len(str1)>len(str2):
            str1,str2=str2,str1        
        m=len(str1)
        n=len(str2)
        res=[0,0]
        for i in range(m):
            if str2.find(str1[i])!=-1:
                for j in range(str2.find(str1[i]),n):
                    k=0
                    while i+k<m and j+k<n and str1[i+k]==str2[j+k]:
                        k+=1
                    if k>res[1]:
                        res[0]=i
                        res[1]=k
        start,end=res[0],res[0]+res[1]
        return str1[start:end]
```
-滑动窗口
```python
import operator
class Solution:
    def LCS(self , str1 , str2 ):
        # write code here       
        m=len(str1)
        n=len(str2)
        res=[0,0]
        def maxLength(i,j,length):
            k=0
            for c in range(length):
                if str1[i+c]==str2[j+c]:
                    k+=1
                    if k>res[1]:
                        res[0]=i+c#此時j发生了递增，i+j指向了公共子串的末尾
                        res[1]=k 
                else:
                    k=0
        for i in range(m):
            temp=min(n,m-i)
            maxLength(i, 0, temp)
        for j in range(n):
            temp=min(m,n-j)
            maxLength(0,j, temp)
        start,end=res[0]-res[1]+1,res[0]+1
        return str1[start:end]
```
