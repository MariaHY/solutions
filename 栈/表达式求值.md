> 二叉树的前序，中序和后序遍历分别对应前缀，中缀和后缀形式。中缀形式的表达式会有歧义，需要使用优先级规则分析中缀表达式。\ 利用操作数栈，从左到右或从右到左扫描表达式。遇到操作数，则压入堆栈，遇到操作符，则与栈顶操作数匹配，将计算结果重新压入操作数栈。\
-根据运算符对数字进行入栈和出栈操作，需先设起始运算符为'+'，后续运算符触发数值变动时，计算的为此时运算符之前的表达式结果。
```python
#
# 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
# 返回表达式的值
# @param s string字符串 待计算的表达式
# @return int整型
#
class Solution:
    def solve(self , s ):
        # write code here
        s=s.strip()
        n=len(s)
        sign='+'
        num=0
        stack=list()
        i=0
        while i<n:
            c=s[i]
            if c!=' ' and c.isdigit():                
                num=num*10+ord(c)-ord('0')
            if c=='(':   
                counter=1
                j=i+1
                while counter!=0:
                    if s[j]=='(':
                        counter+=1
                    if s[j]==')':
                        counter-=1
                    j+=1
                num=self.solve(s[i+1:j-1])
                i=j-1#指向），如果）为末尾，触发stack返回值，否则指针后移直至符合某一判断条件
                continue
            if c in'+-*' or i==n-1:                
                if sign=='+':
                    stack.append(num)
                elif sign=='-':
                    stack.append(-num)
                elif sign=='*':
                    stack.append(stack.pop()*num)#上一个符号为*，此时*后面的数值num已经赋值
                num=0
                sign=c
            i+=1
        return sum(stack)
```

```python
import operator
class Solution:
    def solve(self , s ):
        # write code here
        ope={'+':operator.add,'-':operator.sub,'*':operator.mul}
        s=s.strip()
#         print(s)
        n=len(s)
        sign=list()
        num=list()
        number=0
        flag=False
        for i in s:
            if i.isdigit():
                number=number*10+ord(i)-ord('0')
                flag=True
#                 print('d:', i,'num:',num,'sign:',sign,'flag:',flag)
                continue
            else:
                if flag:
                    num.append(number)
                    number=0
                    flag=False
                if i=='(' or i=='*':
                    sign.append(i)
#                     print('(*:', i, 'num:', num, 'sign:', sign)
                if i==')':
                    while sign[-1]!='(':
                        num.append(sign.pop())
                    sign.pop()
#                     print('):', i,'num:',num,'sign:',sign)
                if i in '+-':
                    while sign and sign[-1] in '+-*':
                        num.append(sign.pop())
                    sign.append(i)
        if flag:
            num.append(number)
            flag=True
#         print('num：',num)
        while sign:
            num.append(sign.pop())
#         print('num：',num)
        res=list()
        for i in num:
            if type(i)==int:
                res.append(i)
            else:
                a =res.pop()
                b =res.pop()
                res.append(ope[i](b,a))
        return res.pop()
```
