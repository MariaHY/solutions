```python
#
# 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
# 返回表达式的值
# @param s string字符串 待计算的表达式
# @return int整型
#
class Solution:
    def solve(self , s ):
        # write code here
        s=s.strip()
        n=len(s)
        sign='+'
        num=0
        stack=list()
        i=0
        while i<n:
            c=s[i]
            if c!=' ' and c.isdigit():                
                num=num*10+ord(c)-ord('0')
            if c=='(':   
                counter=1
                j=i+1
                while counter!=0:
                    if s[j]=='(':
                        counter+=1
                    if s[j]==')':
                        counter-=1
                    j+=1
                num=self.solve(s[i+1:j-1])
                i=j-1#指向），如果）为末尾，触发stack返回值，否则指针后移直至符合某一判断条件
                continue
            if c in'+-*' or i==n-1:                
                if sign=='+':
                    stack.append(num)
                elif sign=='-':
                    stack.append(-num)
                elif sign=='*':
                    stack.append(stack.pop()*num)#上一个符号为*，此时*后面的数值num已经赋值
                num=0
                sign=c
            i+=1
        return sum(stack)
```
